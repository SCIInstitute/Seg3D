set(APPLICATION_RENDERER_SHADER_SRCS

  Fog.vert
  Fog.frag
  SliceShader.frag
  SliceShader.vert

  IsosurfaceShader.vert

target_link_libraries(Application_Renderer
  Core_Utils
  Core_EventHandler
  Core_TextRenderer
  Core_Graphics
  Core_RenderResources
  Core_Renderer
  Core_VolumeRenderer
  Application_Viewer
  ${SCI_GLEW_LIBRARY}
  ${SCI_BOOST_LIBRARY}
